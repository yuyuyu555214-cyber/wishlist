<script>

function chg(id, num) {

  let v = Number(document.getElementById(id).textContent);

  v = Math.max(0, v + num);

  document.getElementById(id).textContent = v;

  calc();

}


function calc() {

  const wp = num("wishport"), wa = num("wisharc"), nm = num("normal");

  const sion = num("sion"), riku = num("riku"), yushi = num("yushi");

  const jaehee = num("jaehee"), ryo = num("ryo"), sakuya = num("sakuya");


  const dreamyTotal = sion + riku + yushi + jaehee + ryo + sakuya;


  // トレカ

  set("chemiA", wp * 2);

  set("selfA", wp * 1);

  set("chemiB", wa * 3);

  set("selfC", nm * 1);

  set("selfD", dreamyTotal);


  // 封入シリアル

  const serialInner =

    wp * 3 + wa * 2 + nm * 1 + dreamyTotal * 1;

  set("serialInner", serialInner);


  // 外付けシリアル（各ショップ）

  const serTower = num("tower");

  const serAmazon = num("amazon");

  const serHMV = num("hmv");

  const serCDJ = num("cdjapan");


  set("serTower", serTower);

  set("serAmazon", serAmazon);

  set("serHMV", serHMV);

  set("serCDJ", serCDJ);


  // 金額

  const money =

    wp * 8500 +

    wa * 5500 +

    nm * 2800 +

    dreamyTotal * 3300;


  set("money", money.toLocaleString());

}


function num(id) { return Number(document.getElementById(id).textContent); }

function set(id, value) { document.getElementById(id).textContent = value; }

</script>
